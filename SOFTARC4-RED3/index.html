            using System;

class TemperatureConverter
{
    static void Main(string[] args)
    {
        bool runApp = true;

        while (runApp)
        {
            Console.Clear();
            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("*****************************************");
            Console.ForegroundColor = ConsoleColor.Cyan;
            Console.WriteLine("Welcome to the Temperature Conversion");
            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("*****************************************\n");
            Console.WriteLine("First Converting Fahrenheit to Celsius\n next will be Celsius to Fahrenheit.\n");
            
            Console.WriteLine("Enter a Fahrenheit value to convert to Celsius:\n");
            Console.ForegroundColor = ConsoleColor.Green;
            double fahrenheit = GetUserInput();
            double celsius = ConvertFahrenheitToCelsius(fahrenheit);

            Console.ForegroundColor = ConsoleColor.Red;
            Console.WriteLine($"{fahrenheit} degrees Fahrenheit\n");
            Console.ResetColor();
            Console.ForegroundColor = ConsoleColor.Cyan;
            Console.WriteLine("equivalent to\n");
            Console.ForegroundColor = ConsoleColor.Blue;
            Console.WriteLine($"{celsius} degrees Celsius\n");
            Console.ResetColor();

            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("*****************************************\n");

            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("Enter a Celsius value to convert to Fahrenheit:\n");
            Console.ForegroundColor = ConsoleColor.Green;
            double celsius2 = GetUserInput();
            double fahrenheit2 = ConvertCelsiusToFahrenheit(celsius2);

            Console.ForegroundColor = ConsoleColor.Blue;
            Console.WriteLine($"{celsius2} degrees Celsius\n");
            Console.ForegroundColor = ConsoleColor.Cyan;
            Console.WriteLine("equivalent to\n");
            Console.ForegroundColor = ConsoleColor.Red;
            Console.WriteLine($"{fahrenheit2} degrees Fahrenheit\n");
            Console.ResetColor();

            
            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("*****************************************\n");
            
            Console.WriteLine("Do you want to continue? (yes/no)");
            string? userResponse = Console.ReadLine();

            if (userResponse == "no")
            {
                runApp = false;
            }

            Console.ResetColor();
        }
    }

    static double GetUserInput()
    {
        double userInput;

        while (true)
        {
            Console.Write("Enter a number: ");

            try
            {
                userInput = Convert.ToDouble(Console.ReadLine());
                break;
            }
            catch (FormatException)
            {
                // If user type a letters in converter.
                Console.ForegroundColor = ConsoleColor.DarkRed;
                Console.WriteLine("Please enter a number.");
                Console.WriteLine();
            }
        }

        return userInput;
    }
    static double ConvertFahrenheitToCelsius(double fahrenheit)
    {
        double celsius = (int)Math.Round((fahrenheit - 32) * 5 / 9);
        return celsius;
    }

    static double ConvertCelsiusToFahrenheit(double celsius)
    {
        double fahrenheit = (int)Math.Round((celsius * 9 / 5) + 32);
        return fahrenheit;
    }

}
